Introduction

This is smooth-validator for Javascript frontend and backend. easy to use in any kind of data validation.

Installation

We can easily downlaod and install latest version 3.0.0 via npm.

Command: npm install smooth-validator

General

This Validators are applicable for all kinds of data types.

Required

This is useful when you dont want user to allow empty value

var validate = parser({ id: 'required' })

var res = validate({ id : -24.6778 });
console.log( res.message) //"validation passed"

var res = validate({});
console.log( res.message) //"id is required"

Same

If you want to variables to have same value you can use `same` (like confirm_passowrd).

var validate = parser({ confirm_passowrd: 'same:password' })

var res = validate({ password : 'strong', confirm_passowrd: 'strong' });
console.log( res.message) //"validation passed"

var res = validate({ password : 'strong', confirm_passowrd: 'week' });
console.log( res.message) //"confirm_passowrd is not equal to password"

Integer

Various validation for integer data types

Numeric

To verify given input in numeric

var validate = parser({ id: 'numeric' })

var res = validate({ id : '-24.6778' });
console.log( res.message) //"validation passed"

var res = validate({ id : '-24. 6778'});
console.log( res.message) //"id is not number"

Between

Set a range for input integer.

var validate = parser({ id: 'between:10,30' })

var res = validate({ id : 10 });
console.log( res.message) //"validation passed"

var res = validate({ id : 9 });
console.log( res.message) //"id should be greater than 10 and less than 30"

Greater Than

Confirm that given input is greater than set value or variable.

var validate = parser({ id: 'gt:marks' })
var res = validate({ id : 10.1, marks: 10 });
console.log( res.message) //"validation passed"

var validate = parser({ id: 'gt:10' })
var res = validate({ id : 10 });
console.log( res.message) //"id should be greater than 10"

Greater Than / Equals

Confirm that given input is greater than or equals to set value or variable.

var validate = parser({ id: 'gte:10' })
var res = validate({ id : 10 });
console.log( res.message) //"validation passed"

var validate = parser({ id: 'gte:marks' })
var res = validate({ id : 9, marks: 10 });
console.log( res.message) //"id should be greater than or equals to marks"

Less Than

Confirm that given input is less than set value or variable.

var validate = parser({ id: 'lt:marks' })
var res = validate({ id : 9.1, marks: 10 });
console.log( res.message) //"validation passed"

var validate = parser({ id: 'lt:10' })
var res = validate({ id : 10 });
console.log( res.message) //"id should be less than 10"

Less Than / Equals

Confirm that given input is greater than or equals to set value or variable.

var validate = parser({ id: 'lte:10' })
var res = validate({ id : 10 });
console.log( res.message) //"validation passed"

var validate = parser({ id: 'lte:marks' })
var res = validate({ id : 10.1, marks: 10 });
console.log( res.message) //"id should be less than or equals to marks"

String

Various validation for string data types

Min

To verify given input has minimum length

var validate = parser({ password: 'min:6' })
var res = validate({ password : 'passing' });
console.log( res.message) //"validation passed"

var validate = parser({ password: 'min:8' })
var res = validate({password : 'passing'});
console.log( res.message) //"password's length is less than 8"

Max

To verify given input has maximum length

var validate = parser({ password: 'max:8' })
var res = validate({ password : 'passing' });
console.log( res.message) //"validation passed"

var validate = parser({ password: 'max:6' })
var res = validate({password : 'passing'});
console.log( res.message) //"password's length is greater than 6"

Size

To verify given input has specific length

var validate = parser({ password: 'size:7' })
var res = validate({ password : 'passing' });
console.log( res.message) //"validation passed"

var validate = parser({ password: 'size:6' })
var res = validate({password : 'passing'});
console.log( res.message) //"password's length not equal to 6"

In

To verify given input value is belongs to specific array

var validate = parser({ password: 'in:passing,failing' })

var res = validate({ password : 'passing' });
console.log( res.message) //"validation passed"

var res = validate({password : 'pass'});
console.log( res.message) //"password's value is not in array"

Not In

To verify given input value is not belongs to specific array

var validate = parser({ status: 'not_in:inactive,delete' })

var res = validate({ status : 'active' });
console.log( res.message) //"validation passed"

var res = validate({ status : 'inactive' });
console.log( res.message) //"status's value is in invalid array"

Alphabetic

To verify given input value is alphabetic

var validate = parser({ name: 'alpha' })
var res = validate({ name : 'bhagyesh' });
console.log( res.message) //"validation passed"

var validate = parser({ name: 'alpha:dash,space,underscope' })
var res = validate({ name : 'bhagy _e-sh' });
console.log( res.message) //"validation passed"

var validate = parser({ name: 'alpha' })
var res = validate({ name : 'bhagy _e-sh10' });
console.log( res.message) //"name does not match alpha"
